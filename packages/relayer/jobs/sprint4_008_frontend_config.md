```typescript
import react from '@vitejs/plugin-react'

// https://vitejs.dev/config/

  plugins: [react()],
  server: {
    port: 3000,
    host: true,
  },
  build: {
    outDir: 'dist',
    sourcemap: true,
  },
  define: {
    // Required for some wallet libraries
    'process.env': {},
    global: 'globalThis',
  },
})
```
